function checkSquence (arr) {
                if (arr[0] === arr[1]){
                        return true;
                
                }
                if (arr[1] === arr[2]){
                        return true;
                
                }
                if (arr[2] === arr[3]){
                        return true;
                
                }
                if (arr[3] === arr[4]){
                        return true;
                
                }
                if (arr[4] === arr[5]){
                    return true;
                }
                return false;
            }
            
            function checkIncrease (arr) {
                if (parseInt(arr[0]) <= parseInt(arr[1]) && 
                    parseInt(arr[1]) <= parseInt(arr[2]) &&
                    parseInt(arr[2]) <= parseInt(arr[3]) &&
                    parseInt(arr[3]) <= parseInt(arr[4]) &&
                    parseInt(arr[4]) <= parseInt(arr[5])
                    ) {
                    return true;
                }
                return false;
            }
            
            function password (min, max) {
                let currentNum = min;
                let valids = []
                let validpw = 0;
                for (let current = currentNum; current <= max; current++) {
                    let numString = current.toString();
                    let digits = numString.split("");
                    let validseq = checkSquence(digits);
                    let validinc = checkIncrease(digits);
                    if (validseq && validinc){
                        valids.push(current);
                        validpw = validpw + 1;
                    }
                }
                console.log(validpw)
            }
